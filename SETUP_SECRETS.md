#!/bin/bash
# Script para configurar secrets no GitHub
# Este é um guia - execute os comandos manualmente ou via GitHub CLI

echo "🔐 Configurando GitHub Secrets para Deploy"
echo ""
echo "Opção 1: Via GitHub CLI (recomendado)"
echo "==========================================="
echo ""
echo "# Instalar GitHub CLI (se não tiver)"
echo "brew install gh"
echo ""
echo "# Login"
echo "gh auth login"
echo ""
echo "# Adicionar secrets"
echo "gh secret set DEPLOY_HOST --body '192.168.1.100'  # IP da sua KVM"
echo "gh secret set DEPLOY_USER --body 'aprova-deploy'  # Usuário SSH"
echo "gh secret set DEPLOY_PORT --body '22'             # Porta SSH"
echo ""
echo "# Para a chave SSH privada:"
echo "gh secret set DEPLOY_KEY < ~/.ssh/id_rsa"
echo ""
echo "# Stripe webhook secret (para notificações)"
echo "gh secret set STRIPE_WEBHOOK_TEST --body 'whsec_test_...'  # Para testes"
echo ""
echo "---"
echo ""
echo "Opção 2: Via Dashboard GitHub"
echo "==========================================="
echo "1. Acesse: https://github.com/xrrac42/aprova-cards-v2/settings/secrets/actions"
echo "2. Clique 'New repository secret'"
echo "3. Adicione:"
echo ""
echo "   Name: DEPLOY_HOST"
echo "   Value: [IP ou hostname da sua KVM]"
echo ""
echo "   Name: DEPLOY_USER"
echo "   Value: [usuário SSH, ex: root ou deploy]"
echo ""
echo "   Name: DEPLOY_PORT"
echo "   Value: [porta SSH, geralmente 22]"
echo ""
echo "   Name: DEPLOY_KEY"
echo "   Value: [conteúdo completo do ~/.ssh/id_rsa]"
echo ""
echo "---"
echo ""
echo "✅ Após configurar os secrets, o deploy automático funcionará!"
